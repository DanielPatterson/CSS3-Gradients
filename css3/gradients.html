CSS3 Gradients

Overview

Create smooth transitions with multiple colors.

<ul class="demo list-unstyled list-inline">
    <li class="default"></li>
    <li class="ltr"></li>
    <li class="rtl"></li>
    <li class="linear-default"></li>
    <li class="linear-reverse"></li>
    <li class="linear"></li>
</ul>

.default {
background: #374252; /* Fallback */
background: -webkit-linear-gradient(#374252, #4C6575); /* Safari */
background: -o-linear-gradient(#374252, #4C6575); /* Opera */
background: -moz-linear-gradient(#374252, #4C6575); /* Firefox */
background: linear-gradient(#374252, #4C6575); /* default */
}

.ltr {
background: #374252; /* Fallback */
background: -webkit-linear-gradient(left, #374252, #4C6575); /* Safari */
background: -o-linear-gradient(right, #374252, #4C6575); /* Opera */
background: -moz-linear-gradient(right, #374252, #4C6575); /* Firefox */
background: linear-gradient(to right, #374252, #4C6575); /* default */
}

.rtl {
background: #374252; /* Fallback */
background: -webkit-linear-gradient(right, #374252, #4C6575); /* Safari */

background: -moz-linear-gradient(left, #374252, #4C6575); /* Firefox */

} 

.linear-default {

} 

.linear-reverse {

.linear {
